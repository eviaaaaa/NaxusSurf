# 如何记录一个Act

## 技术方案概览

1. **依赖库**: 使用 `python-pptx` 生成 PowerPoint 文件。
2. **文件管理**: 为每次任务创建一个唯一的 Session 文件夹（例如 `data/traces/task_20231027_1015/`），用于存放过程截图。
3. **数据流**:
    * **Loop 开始**: 截图（记录当前看到的状态）。
    * **LLM 决策**: 记录 Agent 的思考 (Thought) 和决定做的动作 (Action)。
    * **Loop 结束**: 将截图路径和文本描述存入列表。
    * **任务结束**: 调用 PPT 生成器，遍历列表生成幻灯片。

### 2. 模块实现

我们需要新建一个工具模块 `utils/report_generator.py`，专门负责把图片和文字变成 PPT。

#### A. PPT 生成器 (`utils/report_generator.py`)

```python
import os
from pptx import Presentation
from pptx.util import Inches, Pt
from pptx.enum.text import PP_ALIGN

class PPTReporter:
    def __init__(self, task_id, output_dir="data/reports"):
        self.task_id = task_id
        self.output_dir = output_dir
        self.steps = []  # 存储每一步的数据: { 'image': path, 'thought': str, 'action': str }
        
        if not os.path.exists(output_dir):
            os.makedirs(output_dir)

    def add_step(self, image_path, thought, action):
        """记录单步操作"""
        self.steps.append({
            "image": image_path,
            "thought": thought,
            "action": action
        })

    def generate_ppt(self, filename="report.pptx"):
        """生成最终的 PPT 文件"""
        prs = Presentation()
        
        # 1. 封面页
        title_slide_layout = prs.slide_layouts[0]
        slide = prs.slides.add_slide(title_slide_layout)
        title = slide.shapes.title
        subtitle = slide.placeholders[1]
        title.text = f"NexusSurf Agent 执行报告"
        subtitle.text = f"Task ID: {self.task_id}\nTotal Steps: {len(self.steps)}"

        # 2. 内容页 (每一步一页)
        # 使用空白布局，灵活排版
        blank_slide_layout = prs.slide_layouts[6] 

        for index, step in enumerate(self.steps):
            slide = prs.slides.add_slide(blank_slide_layout)
            
            # --- 标题 (步骤数) ---
            left = Inches(0.5)
            top = Inches(0.2)
            width = Inches(9)
            height = Inches(0.5)
            txBox = slide.shapes.add_textbox(left, top, width, height)
            tf = txBox.text_frame
            tf.text = f"Step {index + 1}: {step['action']}"
            tf.paragraphs[0].font.bold = True
            tf.paragraphs[0].font.size = Pt(24)

            # --- 图片 (居中放置) ---
            img_path = step['image']
            if os.path.exists(img_path):
                # 设定图片位置和大小 (根据 PPT 标准 16:9 设置)
                left = Inches(0.5)
                top = Inches(1.0)
                height = Inches(4.5) # 固定高度，宽度自适应
                slide.shapes.add_picture(img_path, left, top, height=height)

            # --- 思考过程 (底部文本框) ---
            left = Inches(0.5)
            top = Inches(5.8)
            width = Inches(9)
            height = Inches(1.5)
            
            txBox = slide.shapes.add_textbox(left, top, width, height)
            text_frame = txBox.text_frame
            text_frame.word_wrap = True
            
            p = text_frame.add_paragraph()
            p.text = f"Agent Thought: {step['thought']}"
            p.font.size = Pt(14)

        # 3. 保存文件
        save_path = os.path.join(self.output_dir, filename)
        prs.save(save_path)
        print(f"[Report] PPT generated at: {save_path}")
        return save_path
```

#### B. 集成到主控 Agent (`core/agent.py`)

我们需要在 OODA 循环中插入截图和记录逻辑。

```python
import time
import os
from utils.report_generator import PPTReporter

class BrowserAgent:
    def __init__(self, browser_manager, vector_store):
        # ... 初始化代码 ...
        self.task_id = f"task_{int(time.time())}"
        self.screenshot_dir = f"data/traces/{self.task_id}"
        
        # 初始化报告器
        self.reporter = PPTReporter(task_id=self.task_id)
        
        if not os.path.exists(self.screenshot_dir):
            os.makedirs(self.screenshot_dir)

    def run(self, user_objective):
        step_count = 0
        
        while step_count < 20:
            # =================================================
            # 1. 观察 (Observe) & 截图 (Capture)
            # =================================================
            # 在决策之前截图，记录 Agent 看到的状态
            screenshot_path = os.path.join(self.screenshot_dir, f"step_{step_count:03d}.png")
            
            # 调用 Playwright 截图 (假设 browser_manager 暴露了 page 对象)
            # full_page=False 更像 PPT 展示，focus 在视口
            self.browser.page.screenshot(path=screenshot_path)
            
            # 获取 DOM
            clean_html, id_map = self.browser.get_current_state()

            # =================================================
            # 2. 决策 (Decide)
            # =================================================
            # ... 调用 LLM 获取 action_response ...
            # 假设 LLM 返回结构: 
            # { "thought": "我需要点击搜索框...", "tool": "click", "args": {"id": "15"} }
            
            llm_decision = self._call_llm(clean_html, user_objective)
            
            current_thought = llm_decision.get('thought', 'No thought provided')
            current_action_desc = f"{llm_decision.get('tool')} -> {llm_decision.get('args')}"

            # =================================================
            # 3. 记录 (Record)
            # =================================================
            # 将这一步的信息添加到报告器
            self.reporter.add_step(
                image_path=screenshot_path,
                thought=current_thought,
                action=current_action_desc
            )

            # =================================================
            # 4. 行动 (Act)
            # =================================================
            if llm_decision.get('tool') == 'FINISH':
                print("任务完成！")
                break
                
            self._execute_tool(llm_decision)
            
            # 等待页面加载以便下一次循环
            self.browser.page.wait_for_load_state('networkidle')
            step_count += 1

        # =================================================
        # 5. 任务结束：生成 PPT
        # =================================================
        ppt_path = self.reporter.generate_ppt(filename=f"{self.task_id}_report.pptx")
        return f"Task Finished. Report saved at {ppt_path}"
```

### 3. 优化建议 (进阶版)

为了让 PPT 看起来更专业，你可以做以下微调：

1. **视觉高亮 (Visual Highlighting)**:
    在截图 **之后** 但在 **保存** 之前，利用 Playwright 的 `evaluate` 功能，给即将操作的元素加一个红框，然后再截图。
    *这样 PPT 里就能清楚地看到 Agent 这一步点了哪里。*

    ```python
    # 修改核心循环
    
    # 1. LLM 决定要点 ID 为 15 的元素
    target_id = llm_decision['args'].get('id')
    
    # 2. 高亮该元素 (JS 注入)
    if target_id:
        selector = self.browser.current_id_map[target_id]
        self.browser.page.evaluate(f"""
            const el = document.querySelector('{selector}');
            if(el) {{
                el.style.border = '5px solid red';
                el.style.boxShadow = '0 0 10px red';
            }}
        """)
    
    # 3. 截图 (带红框)
    self.browser.page.screenshot(path=screenshot_path)
    
    # 4. 清除高亮 (为了不影响下一步观察，或者直接刷新/跳转也行)
    # ...
    
    # 5. 执行真正的点击操作
    ```

2. **处理长截图**:
    如果你想记录整个网页而不仅仅是视口，可以将 `page.screenshot(full_page=True)` 设为 True。但在 PPT 中，长图会被压缩得很细长，通常建议使用默认的 Viewport 截图，这样文字清晰度最高。

3. **错误截图**:
    如果在 `try...except` 块中捕获到异常，可以额外截一张图命名为 `step_xx_error.png`，并在 PPT 中新建一页标红显示 "Task Failed Here"，这对调试非常有帮助。

### 总结

通过引入 `PPTReporter` 类并将其挂载到 Agent 的主循环中，你不仅实现了自动化的 Web 操作，还自动生成了可视化的审计报告。这个功能对于向非技术人员（如产品经理或客户）展示 Agent 的工作流程非常有价值。
