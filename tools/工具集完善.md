## 二、需要补充的关键工具

### 1. 视觉与截图工具
```python
class ScreenshotTool(BaseTool):
    """捕获当前页面或特定元素的截图"""
    name = "screenshot"
    description = "捕获当前页面或特定元素的截图，并返回Base64编码的图像数据"
    
    async def _run(self, selector: Optional[str] = None, full_page: bool = False) -> str:
        page = self.async_browser.contexts[0].pages[0]
        if selector:
            element = await page.query_selector(selector)
            if element:
                screenshot = await element.screenshot()
            else:
                raise ValueError(f"元素未找到: {selector}")
        else:
            screenshot = await page.screenshot(full_page=full_page)
        
        # 转换为Base64以便在LLM中使用
        import base64
        return base64.b64encode(screenshot).decode('utf-8')
```

class HoverTool(BaseTool):

class DragAndDropTool(BaseTool):

### 3. 等待与条件工具
```python
class WaitForElementTool(BaseTool):
    """等待特定元素出现"""
    name = "wait_for_element"
    description = "等待指定元素出现、消失或满足特定条件"
    
    async def _run(self, selector: str, state: str = "visible", timeout: int = 30000) -> str:
        page = self.async_browser.contexts[0].pages[0]
        valid_states = ["visible", "hidden", "attached", "detached"]
        if state not in valid_states:
            raise ValueError(f"无效状态，必须是: {valid_states}")
        
        await page.wait_for_selector(selector, state=state, timeout=timeout)
        return f"元素 {selector} 已变为 {state} 状态"
```

### 5. 多标签页/窗口管理工具
```python
class NewTabTool(BaseTool):
    """打开新标签页并切换到它"""
    name = "new_tab"
    description = "打开新标签页，可选择导航到特定URL"
    
    async def _run(self, url: Optional[str] = None) -> str:
        context = self.async_browser.contexts[0]
        new_page = await context.new_page()
        if url:
            await new_page.goto(url)
        return f"已打开新标签页{'并导航到 ' + url if url else ''}"
```

```python
class SwitchTabTool(BaseTool):
    """切换到指定标签页"""
    name = "switch_tab"
    description = "切换到指定索引的标签页(从0开始计数)"
    
    async def _run(self, index: int) -> str:
        context = self.async_browser.contexts[0]
        if index < 0 or index >= len(context.pages):
            raise ValueError(f"无效的标签页索引: {index}, 可用范围: 0-{len(context.pages)-1}")
        
        page = context.pages[index]
        await page.bring_to_front()
        return f"已切换到标签页 {index}"
```

### 6. 文件上传工具
```python
class UploadFileTool(BaseTool):
    """上传文件到页面的文件输入元素"""
    name = "upload_file"
    description = "上传本地文件到指定的文件输入元素"
    
    async def _run(self, selector: str, file_path: str) -> str:
        page = self.async_browser.contexts[0].pages[0]
        input_element = await page.wait_for_selector(selector)
        
        if not os.path.exists(file_path):
            raise FileNotFoundError(f"文件不存在: {file_path}")
        
        await input_element.set_input_files(file_path)
        return f"成功上传文件 {file_path} 到元素 {selector}"
```

### 7. 滚动与视图工具
```python
class ScrollTool(BaseTool):
    """控制页面滚动"""
    name = "scroll"
    description = "滚动页面到指定位置或元素"
    
    async def _run(self, 
                  x: Optional[int] = None, 
                  y: Optional[int] = None,
                  selector: Optional[str] = None,
                  behavior: str = "smooth") -> str:
        page = self.async_browser.contexts[0].pages[0]
        
        if selector:
            element = await page.wait_for_selector(selector)
            await element.scroll_into_view_if_needed()
            return f"已滚动到元素: {selector}"
        
        script = f"""
        window.scrollTo({{
            top: {y if y is not None else 'window.scrollY'},
            left: {x if x is not None else 'window.scrollX'},
            behavior: '{behavior}'
        }});
        """
        await page.evaluate(script)
        return f"已滚动到位置: x={x or 'current'}, y={y or 'current'}"
```

## 四、工具优化建议

1. **统一超时处理**：
```python
class BaseBrowserTool(BaseTool):
    """所有浏览器工具的基类，统一超时和错误处理"""
    playwright_timeout: int = 30000
    
    async def _handle_timeout(self, future, timeout=None):
        try:
            return await asyncio.wait_for(future, timeout=timeout or self.playwright_timeout)
        except asyncio.TimeoutError:
            raise TimeoutError(f"操作超时 ({timeout or self.playwright_timeout}ms)")
```

2. **添加视觉反馈**：
   - 结合截图工具，在关键操作前后自动截图
   - 为 LLM 提供视觉上下文，增强决策能力

3. **智能元素定位**：
```python
class SmartClickTool(ClickTool):
    """增强版点击工具，支持多种定位策略"""
    async def _run(self, target: str) -> str:
        # 尝试多种定位策略
        strategies = [
            lambda: self._click_by_selector(target),
            lambda: self._click_by_text(target),
            lambda: self._click_by_aria_label(target)
        ]
        
        for strategy in strategies:
            try:
                return await strategy()
            except Exception as e:
                continue
        
        raise ValueError(f"无法定位元素: {target}")
```

4. **状态跟踪与上下文管理**：
```python
class BrowserStateTracker:
    """跟踪浏览器状态，为AI提供上下文"""
    def __init__(self, browser: AsyncBrowser):
        self.browser = browser
        self.navigation_history = []
        self.active_tabs = []
    
    async def capture_state(self) -> dict:
        """捕获当前浏览器状态"""
        page = self.browser.contexts[0].pages[0]
        return {
            "url": page.url,
            "title": await page.title(),
            "screenshot": await self._get_screenshot(),
            "visible_text": await self._get_visible_text()
        }
```

## 五、实现考虑因素

1. **错误处理**：
   - 所有工具应有统一的异常处理
   - 提供有意义的错误信息给 LLM
   - 区分可恢复和不可恢复的错误

2. **性能优化**：
   - 批量操作减少来回通信
   - 智能缓存页面状态
   - 限制截图分辨率以减少数据量

3. **安全考虑**：
   - 限制执行的 JavaScript 代码
   - 验证文件上传路径
   - 防止跨站脚本攻击

4. **状态一致性**：
   - 确保工具操作后更新内部状态
   - 维护导航历史
   - 跟踪活跃标签页

通过补充这些工具，您的 AI 浏览器操作将更加全面、可靠，能够处理复杂的网页交互场景，同时为 LLM 提供更丰富的上下文信息，大幅提升自动化能力。
